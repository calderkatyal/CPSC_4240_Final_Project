# Recommended cluster stack for RTX 5000 Ada:
#   module load CUDA/12.1.1

# We benchmark the official FlashAttention-1 forward kernel inside the same
# Python environment as the local benchmark. CUDA 12.1.1 is a clean choice
# because official FA1 requires CUDA 11.4+ and checks that nvcc matches
# PyTorch's CUDA.
#
# Shared environment bootstrap:
#   python -m venv venv
#   source venv/bin/activate
#   pip install --upgrade pip
#   pip install --index-url https://download.pytorch.org/whl/cu121 torch==2.2.2
#   pip install -r requirements.txt
#
# Clone the official FA1 source tree with submodules:
#   mkdir -p external
#   cd external
#   git clone --branch v1.0.9 --depth 1 --recurse-submodules \
#       https://github.com/Dao-AILab/flash-attention.git flash-attn-fa1
#   cd ..
#
# Build the official FA1 extension in place:
#   cd external/flash-attn-fa1 && python setup.py build_ext --inplace
#
# Benchmark it from the shared environment:
#   cd /path/to/project
#   python python/benchmark_official_flash_attn.py --source-dir external/flash-attn-fa1
