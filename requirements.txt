# Python dependencies for the simplified FlashAttention project.
#
# Recommended cluster stack for RTX 5000 Ada:
#   module load CUDA/12.1.1
#
# We benchmark official FlashAttention-1 and FlashAttention-2 inside one shared
# Python environment. CUDA 12.1.1 is the cleanest common choice because:
#   - official FA1 requires CUDA 11.4+ and checks that nvcc matches PyTorch's CUDA
#   - official FA2 requires CUDA 12.0+
#   - cu121 PyTorch wheels match the CUDA/12.1.1 module well
#
# Shared environment bootstrap:
#   python -m venv venv
#   source venv/bin/activate
#   pip install --upgrade pip
#   pip install --index-url https://download.pytorch.org/whl/cu121 torch==2.2.2
#   pip install -r requirements.txt
#
# Clone the official FA1 source tree with submodules:
#   mkdir -p external
#   cd external
#   git clone --branch v1.0.9 --depth 1 --recurse-submodules \
#       https://github.com/Dao-AILab/flash-attention.git flash-attn-fa1
#   cd ..
#
# Install the official baselines:
#   cd external/flash-attn-fa1 && python setup.py build_ext --inplace
#   pip install "https://github.com/Dao-AILab/flash-attention/releases/download/v2.5.6/flash_attn-2.5.6+cu122torch2.2cxx11abiFALSE-cp310-cp310-linux_x86_64.whl"
#
# Benchmark them from the shared environment:
#   cd /path/to/project
#   python python/benchmark_official_flash_attn.py --version fa1 --source-dir external/flash-attn-fa1
#   python python/benchmark_official_flash_attn.py --version fa2
#
# Note:
#   Official FA1 and FA2 both use the top-level Python package name `flash_attn`.
#   In this project, FA1 is imported from a checked-out source tree via
#   `--source-dir`, while FA2 is imported from the installed wheel.
